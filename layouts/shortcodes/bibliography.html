"cited" }} {{- $citedOnly = true }} {{- else }} {{- $src = . }} {{ end }} {{ end }} {{- if eq (.Get 1) "cited" }}{{- $citedOnly = true }}{{ end }} {{- end }} {{- if $citedOnly }} {{ $references = $.Page.Scratch.Get "citedBib" }} {{- else }} {{- $bibliographyPath := "" }} {{- $pageResource := $.Page.Resources.GetMatch "*bib*.json" -}} {{- if gt (len $src) 1 }} {{- $bibliographyPath = $src }} {{- else if $.Page.Params.bibFile }} {{- $bibliographyPath = $.Page.Params.bibFile }} {{- else if $pageResource }} {{- $bibliographyPath = $pageResource.RelPermalink }} {{- end }} {{- with $bibliographyPath }} {{- $file := os.ReadFile . }} {{- $references = $file | transform.Unmarshal }} {{- end }} {{- end }} {{- $bibParams := dict "references" $references }} {{- partial "bibliography/bibliography-list.html" $bibParams }}
